Dada una lista circular con los datos de los socios de un club: número, nombre y condición ( ´A´= alta, ´B´= baja);
hacer un algoritmo que cree otra lista simplemente encadenada con los socios múltiplos de 100, que no estén dados de
baja y eliminar de la lista original los socios dados de baja ( cond = ´B´).

ACCION 4.4(PRIM1: puntero a SOCIO) ES
    AMBIENTE
        SOCIO = REGISTRO
            num: entero
            nombre: AN(50)
            condicion: ("A","B")
            prox: puntero a SOCIO
        FIN_REGISTRO
        P, AuxP, ANT: puntero a SOCIO

        SALIDA = REGISTRO
            número: entero
            nombre: AN(50)
            prox: puntero a SALIDA
        FIN_REGISTRO
        S, PRIM2: puntero a SALIDA

        PROCEDIMIENTO ELIMINAR_B() ES
            SI AuxP = PRIM1 ENTONCES
                SI AuxP.prox = AuxP ENTONCES
                    PRIM1:= NIL
                SINO
                    SI ANT = NIL ENTONCES
                        ANT:= PRIM1
                        MIENTRAS *ANT.prox <> PRIM1 HACER
                            ANT:= *ANT.prox
                        FIN_MIENTRAS
                    FIN_SI
                    *ANT.prox:= *PRIM1.prox
                    PRIM1:= *PRIM1.prox
                FIN_SI
            SINO
                *ANT.prox:= P
            FIN_SI
            DISPONER(AuxP)
        FIN_PROCEDIMIENTO

        PROCEDIMIENTO NodoAltaMult100() ES
            NUEVO(S)
            *S.nombre:= *AuxP.nombre
            *S.num:= *AuxP.num
            *S.prox:= PRIM2
            PRIM2:= S
        FIN_PROCEDIMIENTO
    PROCESO
        PRIM2:= NIL
        ANT:= NIL
        AuxP:= PRIM1
        P:= AuxP.prox

        MIENTRAS P <> PRIM1 HACER
            SI *AuxP.condicion = "B" ENTONCES
                ELIMINAR_B()
            SINO
                ANT:= AuxP
                SI *AuxP.num mod 100 = 0 ENTONCES
                    NodoAltaMult100()
                FIN_SI
            FIN_SI
            AuxP:= P
            P:= P.prox
        FIN_MIENTRAS

        SI *AuxP.condicion = "B" ENTONCES
            ELIMINAR_B()
        SINO
            SI *AuxP.num mod 100 = 0 ENTONCES
                NodoAltaMult100()
            FIN_SI
        FIN_SI
FIN_ACCION
