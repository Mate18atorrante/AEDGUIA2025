Dada una lista circular con los datos de los socios de un club: número, nombre y condición ( ´A´= alta, ´B´= baja);
hacer un algoritmo que cree otra lista simplemente encadenada con los socios múltiplos de 100, que no estén dados de
baja y eliminar de la lista original los socios dados de baja ( cond = ´B´).

ACCION 4.4(PRIM1: puntero a SOCIO) ES
    AMBIENTE
        SOCIO = REGISTRO
            num: entero
            nombre: AN(50)
            condicion: ("A","B")
            prox: puntero a SOCIO
        FIN_REGISTRO
        P, AuxP, ANT, ULT: puntero a SOCIO

        SALIDA = REGISTRO
            número: entero
            nombre: AN(50)
            prox: puntero a SALIDA
        FIN_REGISTRO
        S, PRIM2: puntero a SALIDA
    PROCESO
        P:= PRIM1
        PRIM2:= NIL
        ANT:= NIL
        ULT:= NIL

        SI PRIM1 <> NIL ENTONCES
            REPETIR
                SI *P.condición = "A" y *P.num mod 100 = 0 ENTONCES
                    NUEVO(S)
                    *S.nombre:= *P.nombre
                    *S.num:= *P.num
                    *S.prox:= PRIM2
                    PRIM2:= S
                FIN_SI

                P:= *P.prox
            HASTA P = PRIM1

            //vuelvo a recorrer la lista de entrada para eliminar los nodos con condicion = "B"
            //ANT en este caso apunta al nodo anterior, AuxP al nodo actual, y P al siguiente
            REPETIR
                AuxP:= P
                P:= *P.prox

                SI *AuxP.condicion = "B" ENTONCES
                    SI AuxP = PRIM1 ENTONCES
                        SI AuxP.prox = AuxP ENTONCES
                            PRIM1:= NIL
                        SINO
                            ULT:= PRIM1
                            MIENTRAS *ULT.prox <> PRIM1 HACER
                                ULT:= *ULT.prox
                            FIN_MIENTRAS
                            *ULT.prox:= *PRIM1.prox
                            PRIM1:= *PRIM1.prox
                            ANT:= ULT
                        FIN_SI
                    SINO
                        *ANT.prox:= P
                    FIN_SI
                    DISPONER(AuxP)
                SINO
                    ANT:= AuxP
                FIN_SI
            HASTA (P = PRIM1) o (PRIM1 = NIL)
        FIN_SI
FIN_ACCION
